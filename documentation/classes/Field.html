import { IField, IHTMLAttributes } from '../interfaces/ifield';
import { ICustomValidation } from '../interfaces/icustom-validation';
import { IValidator } from '../interfaces/ivalidator';
import { EventEmitter, Input, OnInit } from '@angular/core';
import { FormBuilder, FormControl, FormGroup, ValidatorFn, Validators } from '@angular/forms';
import { FormService } from '../services/form.service';

export class Field implements IField, OnInit {

  private static readonly MSGerrors = {
    'required': () => 'Das ist ein Mussfeld!',
    'maxlength': (param) => `Maximale Länge ${param.requiredLength}!`,
    'minlength': (param) => `Mindestens ${param.requiredLength} Zeichen eingeben!`,
    'max': (param) => `Maximum beträgt ${param.min}!`,
    'min': (param) => `Minimum beträgt ${param.max}!`,
    'pattern': (param) => `Das Pattern: <b>${param.pattern}</b> wurde nicht eingehalten!`,
    'email': () => 'Keine valide Emailadresse'
  };

  types: any[] = [
    {value: 'select', viewValue: 'Select'},
    {value: 'text', viewValue: 'Text'},
    {value: 'checkbox', viewValue: 'Checkbox'},
    {value: 'date', viewValue: 'Datum'}
  ];
  @Input() placeholder = 'test';
  @Input() disabled = false;
  @Input() minLength = 3;
  @Input() maxLength: number;
  @Input() min: number;
  @Input() max: number;
  @Input() required: boolean;
  @Input() form: FormGroup;
  @Input() control: FormControl;
  @Input() name = 'test';
  @Input() hintlabel = 'test';
  @Input() tooltip = 'test';
  @Input() fieldType = 'text';

  constructor(
    public fb: FormBuilder,
    public fs: FormService
  ) {
    // console.log(this.form);
    // console.log('ngoninit', this);
    // console.log(fb);
    // this.hint = this.validators.maxLength ? '{{ ' + this.control.value.length + ' / ' + this.validators.maxLength + '}}' : '';
    // this.hintlabel = 'ich bin eine Testbeschreibung';
    // console.log(FormBuilder);
    if (this.fieldType !== 'text') {
      this.htmlType = this.fieldType;
    }
    if (!this.control) {
      if (this.fs.getConfigByName(this.name)) {
        // console.log('config', this.fs.getConfigByName(this.name, '0'));
        const field = this.fs.getConfigByName(this.name);
        this.setUpConfig(field);
      }
      this.control = this.fs.getFormControl(this.name);
    }
  }

  get Field(): Field {
    return this;
  }

  get customValidation(): ICustomValidation[] {
    return this._customValidation;
  }

  set customValidation(value: ICustomValidation[]) {
    this._customValidation = value;
  }

  get description(): string {
    return this._description;
  }

  set description(value: string) {
    this._description = value;
  }

  get hint(): string {
    return this._hint;
  }

  set hint(value: string) {
    this._hint = value;
  }

  get value(): any {
    return this._value || '';
  }

  set value(value: any) {
    this._value = value;
  }

  get htmlType(): string {
    return this._htmlType;
  }

  set htmlType(value: string) {
    this._htmlType = value;
  }

  get tooltipdelay(): number {
    return this._tooltipdelay;
  }

  set tooltipdelay(value: number) {
    this._tooltipdelay = value;
  }

  get tooltipposition(): string {
    return this._tooltipposition;
  }

  set tooltipposition(value: string) {
    this._tooltipposition = value;
  }

  private _htmlAttribute: IHTMLAttributes;

  get htmlAttribute(): IHTMLAttributes {
    return this._htmlAttribute;
  }

  set htmlAttribute(value: IHTMLAttributes) {
    this._htmlAttribute = value;
  }

  private _validators: IValidator = {};
  @Input() change: EventEmitter<any> = new EventEmitter();


  private _customValidation: ICustomValidation[];
  private _description: string;
  private _hint: string;

  private _htmlType: string;
  private _tooltipdelay: number;
  private _tooltipposition: string;

  get validators(): IValidator {
    if (this._validators !== {}) {
      return this._validators;
    } else {
      return {
        required: this.required,
        maxLength: this.maxLength,
        minLength: this.minLength,
        max: this.max,
        min: this.min
      }
        ;
    }
  }

  private _value: any;

  set validators(value: IValidator) {
    this._validators = value;
  }

  buildValidators() {
    // console.log('buildVali', this);
    const validators: ValidatorFn[] = [];
    if (this.validators) {
      if (this._validators.email) {
        validators.push(Validators.email);
      }
      if (this._validators.max) {
        validators.push(Validators.max(this._validators.max));
      }
      if (this._validators.required) {
        validators.push(Validators.required);
      }
      if (this._validators.min) {
        validators.push(Validators.min(this._validators.min));
      }
      if (this._validators.maxLength) {
        validators.push(Validators.maxLength(this._validators.maxLength));
      }
      if (this._validators.minLength) {
        validators.push(Validators.minLength(this._validators.minLength));
      }
      if (this._validators.pattern) {
        validators.push(Validators.pattern(this._validators.pattern));
      }
    }
    return validators;
  }

  getErrorMeesage() {
    // console.log('geterror', this.control);
    if (this.control) {
      for (const error in this.control.errors) {
        if (this.control.errors.hasOwnProperty(error) && this.control.touched || this.control.dirty) {
          // console.log(error);
          return this.getMeesage(error, this.control.errors[error]);
        }
      }
    }
  }

  getMeesage(type: string, param: any) {
    // console.log(param);
    // console.log('msgs', Field.MSGerrors[type](param));
    return Field.MSGerrors[type](param);
  }

  ngOnInit(): void {
    if (!this.form) {
      this.form = this.fb.group({
        [this.name]: this.control
      });
      // console.log('!form', this.form);
    } else {
      if (!this.form.controls[this.name]) {
        this.form.addControl(
          this.name, this.control
        );
      }
      // console.log('form', this.form);
    }
    // console.log(this.form);
    this.change.subscribe((value) => {
      if (value.type === 'name') {
        this.form = this.fb.group({
          [this.name]: this.control
        });
        // console.log(this.form);
      }
    });
  }

  setUpConfig(config: IField) {
    this.description = config.description;
    this.hint = config.hint;
    this.hintlabel = config.description;
    this.validators = config.validators;
    this.disabled = config.disabled;
    this.required = config.validators ? config.validators.required : null;
    this.value = config.value || '';
    this.placeholder = this.placeholder !== 'test' && this.placeholder !== '' ?
      this.placeholder : this.name;
  }

}
