import { IForm } from '../interfaces/iform';
import { Field } from './field';
import { FormBuilder, FormControl, FormGroup, ValidatorFn, Validators } from '@angular/forms';
import { IField } from '../interfaces/ifield';
import { Input, OnInit } from '@angular/core';
import { IValidator } from '../interfaces/ivalidator';
import { FormService } from '../services/form.service';

export class Form implements IForm, OnInit {
  config: IField[];
  @Input() form: FormGroup;
  name: string;

  constructor(
    public fb: FormBuilder,
    public fs: FormService
  ) {
    if (!this.form) {
      this.form = new FormGroup({});
    }
  }

  ngOnInit() {
  }

  buildForm(config: IField[], name: string) {
    this.config = config;
    for (const field of config) {
      this.buildControl(field);
      console.log(field);
    }
    this.fs.addForm(this.form);
    this.fs.addConfig(this.config, name);
    // console.log('form', this.form);
    // console.log('form', this.config);
  }

  buildControl(field: IField) {
    const newField = new Field(this.fb, this.fs);
    console.log(newField);

    this.form.addControl(
      field.name, new FormControl(field.value, this.buildValidators(field.validators))
    );
    // this.fs.addForm(this.form);
    console.log(this.form);
  }

  getConfig(name: string): IField {
    if (this.config) {
      return this.config[name];
    }
  }

  getForm(name: string): FormGroup {
    return this.form;
  }

  getFormCtrl(name: string): FormControl {
    if (this.form) {
      return this.form.controls[name] as FormControl;
    }
    return undefined;
  }

  buildValidators(config: IValidator) {
    // console.log('buildVali', this);
    const validators: ValidatorFn[] = [];
    if (config) {
      if (config.email) {
        validators.push(Validators.email);
      }
      if (config.max) {
        validators.push(Validators.max(config.max));
      }
      if (config.required) {
        validators.push(Validators.required);
      }
      if (config.min) {
        validators.push(Validators.min(config.min));
      }
      if (config.maxLength) {
        validators.push(Validators.maxLength(config.maxLength));
      }
      if (config.minLength) {
        validators.push(Validators.minLength(config.minLength));
      }
      if (config.pattern) {
        validators.push(Validators.pattern(config.pattern));
      }
    }
    return validators;
  }
}
