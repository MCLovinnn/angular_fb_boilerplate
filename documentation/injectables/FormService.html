import { Injectable } from '@angular/core';
import { FormBuilder, FormControl, FormGroup } from '@angular/forms';
import { IField } from '../interfaces/ifield';

@Injectable({
  providedIn: 'root'
})
export class FormService {

  forms: FormGroup;
  configs: IField[];

  constructor(private fb: FormBuilder) {
  }

  addForm(form: FormGroup) {
    if (this.forms !== undefined && this.forms !== null) {
      Object.assign(this.forms, form);
    } else {
      this.forms = form;
    }
    console.log('formService', this.forms);
  }

  addConfig(config: IField[], name: string) {
    if (this.configs !== undefined && this.configs !== null) {
      Object.assign(this.configs, config);
    } else {
      this.configs = config;
    }
    console.log('formConfig', this.configs);

  }

  updateConfig(config: IField) {
    for (let item of this.configs) {
      if (item.name === config.name) {
        item = config;
        console.log(item);
        return true;
      }
    }
    return false;
  }

  getConfigByName(name: string, configName?: string, formName?: string) {
    if (this.configs) {
      if (configName && formName) {
        const index = formName + '.' + configName;
        return this.configs[index] as IField;
      } else {
        for (let i = 0; i < this.configs.length; i++) {
          if (this.configs[i].name === name) {
            return this.configs[i] as IField;
          }
        }
      }
    }
  }

  getFormControl(name: string = 'new'): FormControl {
    if (this.forms && this.forms.controls[name]) {
      return this.forms.controls[name] as FormControl;
    } else {
      return new FormControl('');
    }
  }

  getForms() {
    return this.forms.getRawValue();
  }

  getConfigs() {
    return this.configs;
  }
}
